    }

    }

    @Override
    public boolean eliminar(ID id) {
        return almacenamiento.remove(id) != null;
    }

    @Override

        return almacenamiento.size();
    }

    // M√©todos adicionales espec√≠ficos de Map
    public boolean existe(ID id) {
        return almacenamiento.containsKey(id);
    }

    public Set<ID> obtenerTodosIds() {

    }

public class ServicioInventario<T extends Producto<? extends Number>> {

    }

    public Optional<T> buscarProducto(String id) {
        return repositorio.buscarPorId(id);
    }

    public List<T> buscarProductosPorNombre(String nombre) {
        // Usar Stream API para b√∫squeda
        return repositorio.buscarTodos().stream()
            .filter(p -> p.getNombre().toLowerCase().contains(nombre.toLowerCase()))
            .collect(Collectors.toList());
    }

    public List<T> buscarProductosConBajoStock(int umbral) {
        return repositorio.buscarTodos().stream()
            .filter(p -> p.getCantidadStock() < umbral)
            .collect(Collectors.toList());
    }

    public double calcularValorTotalInventario() {
        // Sumar precio * cantidad para todos los productos
        return repositorio.buscarTodos().stream()
            .mapToDouble(p -> p.getPrecio().doubleValue() * p.getCantidadStock())
            .sum();
    }

    // M√©todo gen√©rico con comod√≠n (wildcard)
    public void imprimirProductos(List<? extends Producto<?>> productos) {
        System.out.println("=== LISTA DE PRODUCTOS ===");

    }

    // M√©todo para ordenar productos por precio
    public List<T> obtenerProductosOrdenadosPorPrecio(boolean ascendente) {
